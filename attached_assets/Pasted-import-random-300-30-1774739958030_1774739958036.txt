def run_test(questions_list):

    total = len(questions_list)
    # ترتيب الأسئلة عشوائياً مع الحفاظ على مستوى الصعوبة داخل كل 30
    # تبسيطاً، سنعرض الأسئلة بالترتيب الحالي مع خلط الخيارات
    for i, q in enumerate(questions_list):
        print(f"\nالسؤال {i+1} (مستوى الصعوبة: {q['level']}): {q['text']}")
        # خلط الخيارات
        options = q['options'][:]
        random.shuffle(options)
        for idx, opt in enumerate(options):
            print(f"   {chr(65+idx)}. {opt}")
        answer = input("أدخل حرف الإجابة (A, B, C, D): ").upper()
        # إيجاد الحرف الصحيح بعد الخلط
        correct_letter = chr(65 + options.index(q['correct']))
        if answer == correct_letter:
            print("✅ إجابة صحيحة!")
            score += 1
        else:
            print(f"❌ إجابة خاطئة. الإجابة الصحيحة هي {correct_letter}: {q['correct']}")
    print(f"\n\nالنتيجة النهائية: {score} من {total} ({score/total*100:.2f}%)")
