    def run_exam(self):
        """Runs the interactive exam."""
        print("\n" + "="*80)
        print(" اختبار التحصيلي - مادة الأحياء 2-1 (نظام المسارات) ".center(80))
        print("="*80)
        print("يتكون الاختبار من 120 سؤالاً مقسمة إلى 4 أجزاء.")
        print("أجب بكتابة رقم الخيار (1، 2، 3، أو 4).")
        print("="*80)

        for i, q in enumerate(self.questions):
            print(f"\nالسؤال {i+1}: {q['question']}")
            for idx, opt in enumerate(q['options']):
                print(f"  {idx+1}. {opt}")
            while True:
                try:
                    ans_input = input("إجابتك (رقم الخيار): ")
                    ans_index = int(ans_input) - 1
                    if 0 <= ans_index < 4:
                        user_answer = q['options'][ans_index]
                        self.user_answers.append(user_answer)
                        break
                    else:
                        print("الرجاء إدخال رقم بين 1 و 4.")
                except ValueError:
                    print("الرجاء إدخال رقم صحيح.")

        self.show_results()

    def show_results(self):
        """Shows the final score and summary."""
        correct = 0
        for i, q in enumerate(self.questions):
            if self.user_answers[i] == q['answer']:
                correct += 1

        score_percentage = (correct / 120) * 100
        print("\n" + "="*80)
        print(" نتيجة الاختبار ".center(80))
        print("="*80)
        print(f"عدد الإجابات الصحيحة: {correct} من 120")
        print(f"النسبة المئوية: {score_percentage:.2f}%")
        if score_percentage >= 90:
            print("ممتاز! مستوى تحصيلك عالي جداً.")
        elif score_percentage >= 75:
            print("جيد جداً. أنت على الطريق الصحيح.")
        elif score_percentage >= 60:
            print("جيد. هناك بعض النقاط تحتاج للمراجعة.")
        else:
            print("ننصحك بمراجعة المادة العلمية مرة أخرى والتركيز على النقاط الأساسية.")

        # Optional: Show detailed breakdown
        show_details = input("\nهل تريد عرض تفاصيل الإجابات الخاطئة؟ (نعم/لا): ")
        if show_details.lower() in ['نعم', 'ن', 'yes', 'y']:
            print("\n" + "-"*80)
            for i, q in enumerate(self.questions):
                if self.user_answers[i] != q['answer']:
                    print(f"\nالسؤال {i+1}: {q['question']}")
                    print(f"إجابتك: {self.user_answers[i]} (خطأ)")
                    print(f"الإجابة الصحيحة: {q['answer']}")
